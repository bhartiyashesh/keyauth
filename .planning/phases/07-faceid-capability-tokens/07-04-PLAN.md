```swift
/// Silent-send branch entrypoint. Extracted from the `default:` clause so tests can call it
/// directly without a live WebSocket. Runs on the MainActor (inherited from the enclosing class).
///
/// FIDO-09 silent-send gate:
///   isInWindow (D-07 + lazy check) AND preference ON (D-17) AND resolver returns account
///   AND TOTP generation succeeds → send + toast + early return.
///
/// FIDO-10 fall-through: any check fails → `pendingCodeRequest = request` (existing behavior).
internal func handleDecodedRequest(_ request: CodeRequest) {
    if TrustWindowManager.shared.isInWindow,
       TrustWindowPreference.isEnabled,
       let resolver = accountResolver,
       let account = resolver(request),
       let code = TOTPGenerator.generate(for: account) {
        sendEncryptedCode(code, requestId: request.id,
                          issuer: account.issuer, label: account.label)
        TrustWindowManager.shared.showToast(for: account.issuer)
        return
    }

    pendingCodeRequest = request
}
```

```swift
// Added at file scope AFTER the `final class AccountStore` declaration closes:

extension AccountStore {
    /// Deterministic account resolution for a decoded `CodeRequest`. Mirrors
    /// `CodeApprovalView.onAppear` semantics so FaceID and silent-send paths agree on
    /// which account a request refers to (Pattern 2 — anti-drift).
    ///
    /// Semantics (first match wins):
    ///   1. Exact issuer+label match (when either field is non-empty on the request).
    ///   2. Domain-based matching (only when both issuer AND label are empty):
    ///      a. Single domain match → return it.
    ///      b. Multiple domain matches → return nil (AMBIGUOUS — silent path defers to FaceID).
    ///   3. Single-account fallback (user only has one account configured).
    ///   4. Otherwise → nil (AMBIGUOUS or empty store).
    ///
    /// Returning `nil` is the sanctioned "defer to FaceID" signal — it is NOT an error.
    func resolve(for request: CodeRequest) -> Account? {
        // 1. Exact issuer+label
        if !request.issuer.isEmpty || !request.label.isEmpty {
            return accounts.first { $0.issuer == request.issuer && $0.label == request.label }
        }
        // 2. Domain-based
        if let domain = request.domain, !domain.isEmpty {
            let domainLower = domain.lowercased()
            let matched = accounts.filter { account in
                let issuerLower = account.issuer.lowercased()
                return domainLower.contains(issuerLower)
                    || issuerLower.contains(domainLower.replacingOccurrences(of: ".com", with: ""))
            }
            if matched.count == 1 { return matched[0] }
            if matched.count > 1 { return nil } // AMBIGUOUS — silent path defers
        }
        // 3. Single-account fallback
        if accounts.count == 1 { return accounts[0] }
        // 4. Ambiguous or empty
        return nil
    }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Chrome extension → relay → iOS | Untrusted `CodeRequest.domain` is already decrypted here; we only USE it for display + resolver matching, never as a silent-send gate per D-14/D-15. |
| `accountResolver` closure | Set ONCE by KeyAuthApp.onAppear to a store-local lookup; if nil, silent-send always falls through — degraded safely to per-fetch FaceID. |
| RelayClient @MainActor serialization | `handleDecodedRequest` inherits MainActor isolation; `isInWindow` read + `sendEncryptedCode` + toast assignment are serialized with `revoke()` so T-7-03/T-7-04 races cannot silently send a code after the window has closed. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-7-01 | Spoofing | Stolen-device attacker triggers extension request within 2-min window | mitigate (partial) | `isInWindow` check gates the silent path. D-05 revocation (Plan 07-03) triggers on app background — the attack window shrinks to "device unlocked + app foregrounded + within 2 min of last approval," which is the deliberate trade-off. Toast (D-09) is the user-awareness mitigation even inside that window. |
| T-7-02 | Spoofing / Repudiation | Phishing-origin replay within 2-min window (D-14 explicit trade-off) | accept (with mitigation) | Origin captured in `CodeRequest.domain` is NOT consulted for window scoping (global scope per D-02). | T-7-08 | Information Disclosure | `accountResolver` returns an account object but caller extracts only `issuer`, `label`, and `secret` (via TOTPGenerator) | accept | No change vs today's `CodeApprovalView.approveAndSend` — same logical extraction. No new surface. |
