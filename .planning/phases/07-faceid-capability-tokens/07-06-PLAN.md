struct KeyAuthApp: App {
    @UIApplicationDelegateAdaptor(AppDelegate.self) var appDelegate
    @StateObject private var store = AccountStore()
    @StateObject private var pairingStore = PairingStore.shared
    @StateObject private var icloudState = ICloudStateObserver.shared
    @StateObject private var trustWindow = TrustWindowManager.shared           // NEW (FIDO-01)
    @Environment(\.scenePhase) private var scenePhase
    @State private var isUnlocked = false
    @State private var deviceToken: String?
    @State private var didBootstrapSyncPreference = false
    @State private var didBootstrapTrustWindowPreference = false               // NEW (bootstrap guard)
    @State private var migration: MigrationCoordinator?

    var body: some Scene {
        WindowGroup {
            Group {
                if let migration {
                    if isUnlocked {
                        ContentView()
                            .environmentObject(store)
                            .environmentObject(pairingStore)
                            .environmentObject(icloudState)
                            .environmentObject(migration)
                            .environmentObject(trustWindow)                    // NEW (FIDO-18 plumbing)
                    } else {
                        LockScreenView { isUnlocked = true }
                    }
                } else {
                    Color.clear
                }
            }
            .onAppear {
                if migration == nil {
                    migration = MigrationCoordinator(store: store)
                }
                bootstrapSyncPreferenceOnce()
                bootstrapTrustWindowPreferenceOnce()                           // NEW (FIDO-14/16 + resolver wiring)
                setupAppDelegate()
                requestPushPermissionAndRegister()
                NSUbiquitousKeyValueStore.default.synchronize()
            }
            .onChange(of: scenePhase) { newPhase in
                if newPhase == .active {
                    store.reload()
                    NSUbiquitousKeyValueStore.default.synchronize()
                }
            }
            .onReceive(NotificationCenter.default.publisher(for: UIApplication.willEnterForegroundNotification)) { _ in
                RelayClient.shared.reconnectIfNeeded()
                connectRelayIfPaired()
            }
            .onReceive(NotificationCenter.default.publisher(for: UIApplication.didEnterBackgroundNotification)) { _ in
                isUnlocked = false
                RelayClient.shared.disconnect()
                // NOTE: TrustWindowManager.shared.revoke() is triggered by its OWN
                // subscription to this notification (set up in TrustWindowManager.bootstrap).
                // Do NOT add a duplicate revoke() call here — the manager is the single owner.
            }
        }
    }

    // NEW (FIDO-14/16 + FIDO-09 resolver wiring)
    private func bootstrapTrustWindowPreferenceOnce() {
        guard !didBootstrapTrustWindowPreference else { return }
        didBootstrapTrustWindowPreference = true
        TrustWindowPreference.bootstrap()
        trustWindow.bootstrap()
        // Wire the silent-send account resolver. Captured weakly on `store` because
        // AccountStore is a class — we must not leak it beyond the KeyAuthApp lifetime.
        RelayClient.shared.accountResolver = { [weak store] request in
            return store?.resolve(for: request)
        }
    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `@StateObject` wrapping of `.shared` singletons | SwiftUI keeps a reference without re-initializing — Pitfall 8 documented; the existing code does the same with `PairingStore.shared` and `ICloudStateObserver.shared`, so Phase 7 follows precedent. |
| `RelayClient.shared.accountResolver` closure | One-way write from KeyAuthApp.onAppear; no secret crosses the boundary — the closure only reads `store.accounts` for a lookup and returns an `Account?`. |
| `[weak store]` capture | `AccountStore` is a class; weak capture prevents closure-retained store leak if SwiftUI tears down the app scene — rare but correct. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-7-03 | TOCTOU | Background notification revokes window; `KeyAuthApp` closure ALSO runs on background notification (line 67-72) | mitigate | Both observers fire on the same notification. The documentation comment in the edit makes explicit that KeyAuthApp must NOT duplicate the revoke() call — `TrustWindowManager.bootstrap()` already set up its own subscription in Plan 07-03. Having both paths call `revoke()` would be harmless (revoke is idempotent) but introduces redundant work; keeping the manager as the single owner is cleaner. |
| T-7-12 | Race / Lifecycle | `onAppear` may fire multiple times (SwiftUI behavior on some scenes / iOS versions) | mitigate | The `didBootstrapTrustWindowPreference` guard flag — mirroring `didBootstrapSyncPreference` from Phase 6 — makes the bootstrap idempotent. `TrustWindowManager.bootstrap()` itself is ALSO idempotent (its own `didBootstrap` guard from Plan 07-03). Dual safety. |
| T-7-15 | Information Disclosure | Resolver closure leaks account data | accept (non-exploitable) | Closure is local to the app process; no IPC surface. The account resolved via `store.resolve` is the SAME account that would be resolved via the existing FaceID path — no new disclosure. |
</threat_model>
