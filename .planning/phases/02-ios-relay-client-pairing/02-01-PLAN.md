<threat_model>
## Threat Model

| Threat | Severity | Mitigation |
|--------|----------|------------|
| Private encryption key leaked from memory | HIGH | Key stored in Keychain with `kSecAttrAccessibleAfterFirstUnlock`; raw key bytes zeroed after deriving shared key (stored as SymmetricKey opaque type) |
| Shared symmetric key stored insecurely | HIGH | Derived key stored as raw bytes in Keychain-backed PairingData struct, not UserDefaults or disk |
| Replay attack on encrypted relay messages | MEDIUM | ChaChaPoly nonce is random per `seal()` call (CryptoKit generates fresh nonce); relay forwards opaquely so server cannot replay |
| Man-in-the-middle during pairing QR scan | LOW | QR contains peer public key; X25519 shared secret is unique to that keypair; attacker would need to intercept AND replace QR content |
| WebSocket connection hijacked | LOW | TLS termination at Railway; relay URL uses wss://; roomId is UUID (not guessable) |
</threat_model>

```swift
import Foundation
import CryptoKit

enum CryptoBoxManager {
    static func generateKeyPair() -> Curve25519.KeyAgreement.PrivateKey {
        return Curve25519.KeyAgreement.PrivateKey()
    }

    static func deriveSharedKey(
        privateKey: Curve25519.KeyAgreement.PrivateKey,
        peerPublicKey: Curve25519.KeyAgreement.PublicKey
    ) throws -> SymmetricKey {
        let sharedSecret = try privateKey.sharedSecretFromKeyAgreement(with: peerPublicKey)
        return sharedSecret.hkdfDerivedSymmetricKey(
            using: SHA256.self,
            salt: Data(),
            sharedInfo: Data("KeyAuth-E2E".utf8),
            outputByteCount: 32
        )
    }

    static func seal(_ plaintext: Data, using key: SymmetricKey) throws -> Data {
        let sealedBox = try ChaChaPoly.seal(plaintext, using: key)
        return sealedBox.combined  // nonce(12) + ciphertext + tag(16)
    }

    static func open(_ combined: Data, using key: SymmetricKey) throws -> Data {
        let sealedBox = try ChaChaPoly.SealedBox(combined: combined)
        return try ChaChaPoly.open(sealedBox, using: key)
    }
}
```

```swift
        reload()

    func reload() {
        do {
            pairingData = try loadFromKeychain()
            error = nil
        } catch {
            self.error = error.localizedDescription
            pairingData = nil
        }
    }

    func savePairing(_ pairing: PairingData) throws {
        try saveToKeychain(pairing)
        pairingData = pairing
        error = nil
    }

    func unpair() {
        deleteFromKeychain()
        pairingData = nil
        error = nil
    }
```

**Private Keychain methods** (follow KeychainManager's upsert pattern exactly):

```swift
// MARK: - Keychain

private func baseQuery() -> [String: Any] {
    var query: [String: Any] = [
        kSecClass as String: kSecClassGenericPassword,
        kSecAttrService as String: service,
        kSecAttrAccount as String: pairingKey
    ]
    if let group = accessGroup {
        query[kSecAttrAccessGroup as String] = group
    }
    return query
}

private func saveToKeychain(_ pairing: PairingData) throws {
    let data = try JSONEncoder().encode(pairing)
    let query = baseQuery()

    let status = SecItemCopyMatching(query as CFDictionary, nil)
    if status == errSecSuccess {
        let update: [String: Any] = [kSecValueData as String: data]
        let updateStatus = SecItemUpdate(query as CFDictionary, update as CFDictionary)
        guard updateStatus == errSecSuccess else {
            throw KeychainError.updateFailed(updateStatus)
        }
    } else {
        var insertQuery = query
        insertQuery[kSecValueData as String] = data
        insertQuery[kSecAttrAccessible as String] = kSecAttrAccessibleAfterFirstUnlock
        let insertStatus = SecItemAdd(insertQuery as CFDictionary, nil)
        guard insertStatus == errSecSuccess else {
            throw KeychainError.saveFailed(insertStatus)
        }
    }
}

private func loadFromKeychain() throws -> PairingData? {
    var query = baseQuery()
    query[kSecReturnData as String] = true
    query[kSecMatchLimit as String] = kSecMatchLimitOne

    var result: AnyObject?
    let status = SecItemCopyMatching(query as CFDictionary, &result)

    guard status == errSecSuccess, let data = result as? Data else {
        if status == errSecItemNotFound { return nil }
        throw KeychainError.loadFailed(status)
    }

    return try JSONDecoder().decode(PairingData.self, from: data)
}

private func deleteFromKeychain() {
    let query = baseQuery()
    SecItemDelete(query as CFDictionary)
}
```

Import `Security` for Keychain APIs. Reuse the existing `KeychainError` enum from `KeychainManager.swift` (it is file-scoped but accessible within the module since no access modifier restricts it).

Create `Shared/RelayClient.swift` as a `@MainActor final class: ObservableObject` singleton.

**Structure:**

```swift
import Foundation

@MainActor
final class RelayClient: ObservableObject {
    static let shared = RelayClient()

    enum ConnectionState {
        case disconnected
        case connecting
        case connected
    }

    @Published private(set) var state: ConnectionState = .disconnected
    @Published var pendingCodeRequest: CodeRequest?

    private var webSocketTask: URLSessionWebSocketTask?
    private var session: URLSession?
    private var roomId: String?
    private var deviceToken: String?

    private init() {}
```

**connect method:**
- Takes `roomId: String`, `relayURL: String`, `deviceToken: String?`
- Guards `state == .disconnected`
- Sets `state = .connecting`
- Constructs URL: `"\(relayURL)?roomId=\(roomId)"`
- Creates `URLSession` with default configuration
- Creates `webSocketTask` from session
- Resumes task
- Starts receive loop
- On first successful receive or after a short delay, sends `join` message:
  ```swift
  let joinPayload: [String: String] = deviceToken.map { ["deviceToken": $0] } ?? [:]
  let joinEnvelope = MessageEnvelope(type: "join", payload: joinPayload)
  send(joinEnvelope)
  state = .connected
  ```

**CRITICAL: The connect method must call `receiveLoop()` immediately after `webSocketTask?.resume()`.** The receive loop is the mechanism for detecting connection success/failure. Send the join message inside the connect method after starting the receive loop.

Actually, use a delegate approach: make `RelayClient` conform to `URLSessionWebSocketDelegate` in an extension. The `urlSession(_:webSocketTask:didOpenWithProtocol:)` callback fires when the connection opens -- send the join message there and set `state = .connected`. The `urlSession(_:webSocketTask:didCloseWith:reason:)` callback fires on close -- set `state = .disconnected` there.

```swift
extension RelayClient: URLSessionWebSocketDelegate {
    nonisolated func urlSession(
        _ session: URLSession,
        webSocketTask: URLSessionWebSocketTask,
        didOpenWithProtocol protocol: String?
    ) {
        Task { @MainActor in
            self.state = .connected
            let joinPayload: [String: String] = self.deviceToken.map { ["deviceToken": $0] } ?? [:]
            let joinEnvelope = MessageEnvelope(type: "join", payload: joinPayload)
            self.send(joinEnvelope)
        }
    }

    nonisolated func urlSession(
        _ session: URLSession,
        webSocketTask: URLSessionWebSocketTask,
        didCloseWith closeCode: URLSessionWebSocketTask.CloseCode,
        reason: Data?
    ) {
        Task { @MainActor in
            self.state = .disconnected
        }
    }
}
```

**disconnect method:**
- `webSocketTask?.cancel(with: .goingAway, reason: nil)` (fast close per Pitfall 7)
- Nil out `webSocketTask` and invalidate session
- Set `state = .disconnected`
- Nil out `pendingCodeRequest`

**send method:**
- Encodes `MessageEnvelope` to JSON string via `JSONEncoder`
- Sends via `webSocketTask?.send(.string(json))`
- Error handler logs but does not throw (best-effort, matches relay APNs pattern)

**receiveLoop method (CRITICAL -- must re-invoke after each message):**
```swift
private func receiveLoop() {
    webSocketTask?.receive { [weak self] result in
        Task { @MainActor in
            guard let self else { return }
            switch result {
            case .success(let message):
                self.handleMessage(message)
                self.receiveLoop()  // MUST re-call for next message
            case .failure:
                self.state = .disconnected
            }
        }
    }
}
```

**handleMessage method:**
- For `.string(let text)`: parse as `MessageEnvelope` via JSONDecoder
- If `envelope.type == "joined"`: log success (already handled state in delegate)
- If `envelope.type == "error"`: log the error payload
- For any other type (opaque forwarded message from Chrome extension):
  - Read `envelope.payload["data"]` as base64 string
  - Decode base64 to Data
  - Decrypt using `CryptoBoxManager.open(encrypted, using: sharedKey)` where `sharedKey = PairingStore.shared.sharedKey`
  - Decode decrypted plaintext as `CodeRequest` via JSONDecoder
  - Set `pendingCodeRequest = request` (this triggers the approval sheet in UI)
- Guard all decryption/decode steps with `guard let` returning silently on failure

**registerToken method:**
- Convenience to send `register_token` message:

```swift
func registerToken(_ token: String) {
    self.deviceToken = token
    guard state == .connected else { return }
    let envelope = MessageEnvelope(type: "register_token", payload: ["deviceToken": token])
    send(envelope)
}
```

**sendEncryptedCode method:**
- Takes `code: String` and `requestId: String`
- Encrypts code JSON using `CryptoBoxManager.seal` with `PairingStore.shared.sharedKey`
- Wraps in MessageEnvelope with type `"code_response"` and payload `["data": base64String]`
- Calls `send(envelope)`
```swift
func sendEncryptedCode(_ code: String, requestId: String) {
    guard let sharedKey = PairingStore.shared.sharedKey else { return }
    let responseJSON: [String: String] = ["code": code, "requestId": requestId]
    guard let plaintext = try? JSONEncoder().encode(responseJSON),
          let encrypted = try? CryptoBoxManager.seal(plaintext, using: sharedKey)
    else { return }
    let envelope = MessageEnvelope(
        type: "code_response",
        payload: ["data": encrypted.base64EncodedString()]
    )
    send(envelope)
}
```
