```swift
// MARK: - Account Resolution for Code Requests

/// Resolves a CodeRequest to a specific Account.
/// Priority: (1) accountId exact match, (2) domain + issuer matching, (3) nil (ambiguous)
func resolve(for request: CodeRequest) -> Account? {
    // D-02: Direct lookup by accountId (most reliable)
    if let accountIdStr = request.accountId,
       let uuid = UUID(uuidString: accountIdStr),
       let account = accounts.first(where: { $0.id == uuid }) {
        return account
    }

    // Fallback: domain matching (same logic as CodeApprovalView)
    guard let domain = request.domain, !domain.isEmpty else {
        // No domain and no accountId -- try issuer/label match
        if !request.issuer.isEmpty {
            return accounts.first(where: { $0.issuer.lowercased() == request.issuer.lowercased() })
        }
        return accounts.count == 1 ? accounts.first : nil
    }

    let domainLower = domain.lowercased()
    let matched = accounts.filter { account in
        let issuerLower = account.issuer.lowercased()
        return domainLower.contains(issuerLower)
            || issuerLower.contains(domainLower.replacingOccurrences(of: ".com", with: ""))
    }

    return matched.count == 1 ? matched.first : nil
}
```

```swift
    /// Send account metadata list to paired Chrome extension (D-01).
    /// Called on every WebSocket connect so extension has fresh data.
    /// Only sends id, issuer, label -- secrets NEVER leave the phone.
    func sendAccountList(from store: AccountStore) {
        guard let sharedKey = PairingStore.shared.sharedKey else { return }
        let metadata: [[String: String]] = store.accounts.map {
            ["id": $0.id.uuidString, "issuer": $0.issuer, "label": $0.label]
        }
        let payload: [String: Any] = ["accounts": metadata]
        guard let jsonData = try? JSONSerialization.data(withJSONObject: payload),
              let encrypted = try? CryptoBoxManager.seal(jsonData, using: sharedKey)
        else { return }
        let envelope = MessageEnvelope(
            type: "account_list",
            payload: ["data": encrypted.base64EncodedString()]
        )
        send(envelope)
        print("[RelayClient] Sent account list (\(metadata.count) accounts)")
    }
```
